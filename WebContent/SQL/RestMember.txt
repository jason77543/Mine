DROP TABLE RESTMEMBER;
DROP SEQUENCE RESTMEMBER_SEQ;

CREATE TABLE RESTMEMBER (
RESTID VARCHAR2(60) NOT NULL ,
RESTLISTNO NUMBER(10) NOT NULL ,
RESTPSW VARCHAR2(60) NOT NULL ,
CONSTRAINT RESTMEMBER_RESTLISTNO_FK FOREIGN KEY (RESTLISTNO) REFERENCES RESTLIST (RESTLISTNO),
CONSTRAINT RESTMEMBER_RESTID_PK PRIMARY KEY (RESTID));

CREATE SEQUENCE RESTMEMBER_SEQ
MINVALUE 0
MAXVALUE 99999
INCREMENT BY 1
START WITH 1
NOCACHE
NOORDER
NOCYCLE;
